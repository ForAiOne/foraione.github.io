        // === GÉNÉRATION DES BOUTONS ===
        function generateButtons() {
            jukeboxImage.onload = function() {
                createButtonsWithPositions();
            };
        }

        // Créer les boutons avec les positions
        function createButtonsWithPositions() {
            buttonPositions.forEach(pos => {
                createButton(pos.label, pos.x, pos.y, pos.type);
            });
        }

        // === CRÉATION D'UN BOUTON ===
        function createButton(label, x, y, type) {
            const btn = document.createElement('button');
            btn.className = `jukebox-btn btn-${type}`;
            btn.textContent = label;
            btn.dataset.label = label;
            btn.dataset.type = type;
            
            // Positionner le bouton en pourcentage
            btn.style.left = x + '%';
            btn.style.top = y + '%';
            btn.style.transform = 'translate(-50%, -50%)';
            
            // Ajouter l'événement de clic
            btn.addEventListener('click', (e) => {
                if (editMode) return; // En mode édition, ne pas d'action
                e.stopPropagation(); // Empêcher la propagation à l'image
                handleButtonClick(btn, label, type);
            });
            
            buttonsOverlay.appendChild(btn);
        }

        // === GESTION DU CLIC SUR BOUTON ===
        function handleButtonClick(btn, label, type) {
            if (type === 'letter') {
                // Sélection d'une lettre
                if (selectedLetter === label) {
                    // Désélectionner si on reclique
                    resetSelection();
                } else {
                    // Nouvelle sélection de lettre
                    resetSelection();
                    selectedLetter = label;
                    btn.classList.add('selected', 'pressed');
                }
            } else if (type === 'number' && selectedLetter) {
                // Sélection d'un chiffre après une lettre
                selectedNumber = label;
                btn.classList.add('selected', 'pressed');
                
                // Valider la combinaison
                validateCombination();
            }
        }

        // === VALIDATION DE LA COMBINAISON ===
        function validateCombination() {
            const combination = selectedLetter + selectedNumber;
            const songName = songsList[combination] || 'Piste inconnue';
            
            // Masquer le now playing
            nowPlaying.classList.remove('visible');
            
            // Animer le jukebox
            jukeboxImage.classList.add('jukebox-active');
            glowEffect.classList.add('active');
            
            // Jouer le son mécanique (simulation)
            playMechanicalSound();
            
            // Afficher les informations après le délai
            setTimeout(() => {
                showNowPlaying(combination, songName);
                startPlaying(songName);
            }, 1500);
            
            // Nettoyer l'animation après délai
            setTimeout(() => {
                jukeboxImage.classList.remove('jukebox-active');
                glowEffect.classList.remove('active');
            }, 2500);
        }

        // === AFFICHER NOW PLAYING ===
        function showNowPlaying(code, title) {
            nowPlayingCode.textContent = code;
            nowPlayingTitle.textContent = title;
            nowPlaying.classList.add('visible');
        }

        // === SON MÉCANIQUE (SIMULATION) ===
        function playMechanicalSound() {
            const audioContext = new (window.AudioContext || window.webkitAudioContext)();
            
            const oscillator = audioContext.createOscillator();
            const gainNode = audioContext.createGain();
            
            oscillator.type = 'sawtooth';
            oscillator.frequency.setValueAtTime(60, audioContext.currentTime);
            oscillator.frequency.exponentialRampToValueAtTime(30, audioContext.currentTime + 0.5);
            
            gainNode.gain.setValueAtTime(0.3, audioContext.currentTime);
            gainNode.gain.exponentialRampToValueAtTime(0.01, audioContext.currentTime + 1.5);
            
            oscillator.connect(gainNode);
            gainNode.connect(audioContext.destination);
            
            oscillator.start(audioContext.currentTime);
            oscillator.stop(audioContext.currentTime + 1.5);
        }

        // === RÉINITIALISER LA SÉLECTION ===
        function resetSelection() {
            selectedLetter = null;
            selectedNumber = null;
            
            document.querySelectorAll('.jukebox-btn').forEach(btn => {
                btn.classList.remove('selected', 'pressed');
            });
        }

        // === MODE ÉDITION ===
        function toggleEditMode() {
            editMode = !editMode;
            
            if (editMode) {
                editPanel.classList.add('visible');
                editModeBtn.classList.add('active');
                editModeBtn.textContent = '❌ QUITTER';
                
                // Masquer les instructions
                instructions.style.display = 'none';
                
                // Activer le clic sur l'image
                jukeboxImage.classList.add('editing');
                jukeboxImage.onclick = handleImageClick;
                
                // Nettoyer les boutons existants
                document.querySelectorAll('.edit-marker').forEach(m => m.remove());
                document.querySelectorAll('.jukebox-btn').forEach(b => b.style.display = 'none');
                
                // Réinitialiser les marqueurs d'édition
                buttonPositions.forEach(p => p.edited = false);
                
                updateEditDisplay();
            } else {
                exitEditMode();
            }
        }

        function setEditType(type) {
            editingType = type;
        }

        function exitEditMode() {
            editMode = false;
            editingType = null;
            
            editPanel.classList.remove('visible');
            editModeBtn.classList.remove('active');
            editModeBtn.textContent = '🎯 MODE ÉDITION';
            
            // Afficher les instructions
            instructions.style.display = 'block';
            
            // Désactiver le clic sur l'image
            jukeboxImage.classList.remove('editing');
            jukeboxImage.onclick = null;
            
            // Nettoyer les marqueurs
            document.querySelectorAll('.edit-marker').forEach(m => m.remove());
            
            // Réafficher les boutons
            document.querySelectorAll('.jukebox-btn').forEach(b => b.style.display = 'flex');
            
            updateEditDisplay();
        }

        function handleImageClick(e) {
            if (!editingType) {
                alert('Veuillez sélectionner "Éditer Lettres" ou "Éditer Chiffres"');
                return;
            }
            
            const rect = jukeboxImage.getBoundingClientRect();
            const x = ((e.clientX - rect.left) / rect.width) * 100;
            const y = ((e.clientY - rect.top) / rect.height) * 100;
            
            // Mettre à jour la position
            updatePosition(editingType, x, y);
        }

        function updatePosition(type, x, y) {
            // Trouver le prochain bouton de ce type non édité
            const nextBtn = buttonPositions.find(p => p.type === type && !p.edited);
            
            if (nextBtn) {
                nextBtn.x = x;
                nextBtn.y = y;
                nextBtn.edited = true;
                
                // Afficher un marqueur
                showMarker(x, y, nextBtn.label);
                
                updateEditDisplay();
            } else {
                alert(`Tous les boutons ${type === 'letter' ? 'lettres' : 'chiffres'} ont été positionnés !`);
            }
        }

        function showMarker(x, y, label) {
            const marker = document.createElement('div');
            marker.className = 'edit-marker';
            marker.style.left = x + '%';
            marker.style.top = y + '%';
            marker.textContent = label;
            
            buttonsOverlay.appendChild(marker);
        }

        function updateEditDisplay() {
            let html = '<strong>Positions enregistrées :</strong><br><br>';
            
            html += '<strong>Lettres :</strong><br>';
            buttonPositions.filter(p => p.type === 'letter').forEach(p => {
                const status = p.edited ? '✅' : '❌';
                html += `${p.label}: x=${p.x.toFixed(1)}%, y=${p.y.toFixed(1)}% ${status}<br>`;
            });
            
            html += '<br><strong>Chiffres :</strong><br>';
            buttonPositions.filter(p => p.type === 'number').forEach(p => {
                const status = p.edited ? '✅' : '❌';
                html += `${p.label}: x=${p.x.toFixed(1)}%, y=${p.y.toFixed(1)}% ${status}<br>`;
            });
            
            editPositions.innerHTML = html;
        }
